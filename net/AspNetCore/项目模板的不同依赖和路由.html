private static IMvcCoreBuilder AddControllersWithViewsCore(IServiceCollection services)
{

        .AddViews()
        .AddRazorViewEngine()
        .AddCacheTagHelper();

    AddTagHelpersFrameworkParts(builder.PartManager);

    return builder;
}
